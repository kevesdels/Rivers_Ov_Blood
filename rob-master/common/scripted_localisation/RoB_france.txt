
# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes




##influence in belarussian government
defined_text = { #
	name = GetFrenchStrenghtInWallonia
	text = {
		trigger = {
			OR = {
				original_tag = FRS
				original_tag = FRA
				original_tag = GER
			}
			check_variable = { var = FRS_walonia_french value = 10 compare = greater_than_or_equals }
		}
		localization_key = frs_strength_humongus
	}
	text = {
		trigger = {
			OR = {
				original_tag = FRS
				original_tag = FRA
				original_tag = GER
			}
			check_variable = { var = FRS_walonia_french value = 6 compare = greater_than_or_equals }
			check_variable = { var = FRS_walonia_french value = 10 compare = less_than }

		}
		localization_key = frs_strength_large
	}
	text = {
		trigger = {
			OR = {
				original_tag = FRS
				original_tag = FRA
				original_tag = GER
			}
			check_variable = { var = FRS_walonia_french value = 3 compare = greater_than_or_equals }
			check_variable = { var = FRS_walonia_french value = 6 compare = less_than }

		}
		localization_key = frs_strength_medium
	}
	text = {
		trigger = {
			OR = {
				original_tag = FRS
				original_tag = FRA
				original_tag = GER
			}
			check_variable = { var = FRS_walonia_french value = 0 compare = greater_than_or_equals }
			check_variable = { var = FRS_walonia_french value = 3 compare = less_than }

		}
		localization_key = frs_strength_small
	}
}
