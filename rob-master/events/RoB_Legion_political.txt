### Ukraine
country_event = {
 id = legion_political.1
 desc = legion_political.1.d
 title = "Fate of Ukraine"
 fire_only_once = yes

 mean_time_to_happen = { days = 1 }
 trigger = {
	has_war = no
	UKR = { NOT = { is_puppet_of = ROOT } }
	202 = { is_owned_and_controlled_by = ROOT }
	 tag = LES
   }

###Federation option
 option = {
	name = "Create the Slavic Confederation Member State"
	trigger = {
	has_country_flag = les_confederation
	}
	ai_chance = { factor = 50 }
	puppet = UKR
	every_owned_state = {
	  limit = { 
	   NOT = {
		is_claim_or_core = yes
	   }
		is_core_of = UKR
	  }
	  UKR = { transfer_state = PREV }
	}
   if = { 
	limit = { ROOT = { ANTI_LIBERAL_ALIGNED_GOVERNMENT = yes } }
   UKR = {
	set_cosmetic_tag = LES_gadja_puppet
	} 
   }  
   }

   ###Federation ignore
   option = {
	name = "Continue Military Occupation"
	trigger = {
	has_country_flag = les_confederation
	}
	ai_chance = { factor = 0 }
	add_stability =-0.1
	add_war_support =-0.1
	add_political_power =-100
   }

   ###Ukraine in West Slavia, West slavia not formed yet
   option = {
	name = "Create West Slavic Federation"
	trigger = {
	has_country_flag = les_onlyruseastslav
	NOT = { has_global_flag = LES_westslavformed }
	}
	puppet = UKR
	every_owned_state = {
	  limit = { 
	   NOT = {
		is_claim_or_core = yes
	   }
		is_core_of = UKR
	  }
	  UKR = { transfer_state = PREV }
	}
	UKR = {
		set_cosmetic_tag = LES_westslav
		set_country_flag = LES_westslavformedhere
		}
		set_global_flag = LES_westslavformed 
   }

   ###Ukraine in West Slavia, West slavia has been formed 
   option = {
	name = "Expand our West Slav Ally"
	trigger = {
	has_country_flag = les_onlyruseastslav
	has_global_flag = LES_westslavformed
	}
	every_owned_state = {
		limit = { 
		 NOT = {
		  is_claim_or_core = yes
		 }
		  is_core_of = UKR
		}
		every_country = {
			limit = {
				has_country_flag = LES_westslavformedhere
			}
			transfer_state = PREV
			}
  		}
	}

   ###Unitary/ Ukraine Not in west slavia
   option = {
	name = "Welcome Home, Brothers"
	trigger = {
		OR = {
			has_country_flag = les_unitary
			has_country_flag = les_bigeastslav
		}
	}
	set_country_flag = LES_may_start_integrating_ukr
  }
  
}
